The circuit I created to make the 3 LEDs blink.


I created my schematic digitally because it was easier for me than drawing it.



// the setup function runs once when you press reset or power the board
void setup() {
  // initialize digital pin 8 as an output.
  pinMode(8, OUTPUT);
  // initialize digital pin 9 as an output.
  pinMode(9, OUTPUT);
  // initialize digital pin 10 as an output.
  pinMode(10, OUTPUT);
}

// the loop function runs over and over again forever
void loop() {
  digitalWrite(8, HIGH);   // turn the first LED on (HIGH is the voltage level)
  delay(500);                       // wait for a second
  digitalWrite(8, LOW);    // turn the first LED off by making the voltage LOW
  delay(500);                       // wait for a second
  digitalWrite(9, HIGH);   // turn the second LED on (HIGH is the voltage level)
  delay(750);                       // wait for a second
  digitalWrite(9, LOW);    // turn the second LED off by making the voltage LOW
  delay(750);    
  digitalWrite(10, HIGH);   // turn the third LED on (HIGH is the voltage level)
  delay(1000);                       // wait for a second
  digitalWrite(10, LOW);    // turn the third LED off by making the voltage LOW
  delay(1000);  
}
